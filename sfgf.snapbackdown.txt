Inputs:
	BarSegments(Numeric);

Variables:
	RangeDelta(0.0),
	RangeDelta1(0.0);

sfgf.snapbackdown = False;
RangeDelta = Range / BarSegments;
RangeDelta1 = Range[1] / BarSegments;

If Open >= High - RangeDelta and
Close <= Low + RangeDelta and
Open[1] <= Low[1] + RangeDelta1 and
Close[1] >= High[1] - RangeDelta1 Then
	sfgf.snapbackdown = True;
